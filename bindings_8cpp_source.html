60// The compositions in the gasMixtureInMolePercent array use the following order and must be sent as mole fractions:
61// 0 - PLACEHOLDER
62// 1 - Methane
63// 2 - Nitrogen
64// 3 - Carbon dioxide
65// 4 - Ethane
66// 5 - Propane
67// 6 - Isobutane
68// 7 - n-Butane
69// 8 - Isopentane
70// 9 - n-Pentane
71// 10 - n-Hexane
72// 11 - n-Heptane
73// 12 - n-Octane
74// 13 - n-Nonane
75// 14 - n-Decane
76// 15 - Hydrogen
77// 16 - Oxygen
78// 17 - Carbon monoxide
79// 18 - Water
80// 19 - Hydrogen sulfide
81// 20 - Helium
82// 21 - Argon
83//
84// For example, a mixture (in moles) of 94% methane, 5% CO2, and 1% helium would be (in mole fractions):
85// gasMixtureInMolePercent(1)=0.94, gasMixtureInMolePercent(3)=0.05, gasMixtureInMolePercent(20)=0.01
86enum class GasComponent
87{
88 PLACEHOLDER = 0,
89 METHANE = 1,
90 NITROGEN = 2,
91 CARBON_DIOXIDE = 3,
92 ETHANE = 4,
93 PROPANE = 5,
94 ISOBUTANE = 6,
95 N_BUTANE = 7,
96 ISOPENTANE = 8,
97 N_PENTANE = 9,
98 N_HEXANE = 10,
99 N_HEPTANE = 11,
100 N_OCTANE = 12,
101 N_NONANE = 13,
102 N_DECANE = 14,
103 HYDROGEN = 15,
104 OXYGEN = 16,
105 CARBON_MONOXIDE = 17,
106 WATER = 18,
107 HYDROGEN_SULFIDE = 19,
108 HELIUM = 20,
109 ARGON = 21
110};

346// Helper function to convert a JavaScript array to a C++ vector
357std::vector<double> array_to_vector(const val &js_array)
358{
359 auto length = js_array["length"].as<unsigned>();
360 std::vector<double> result(length);
361 for (unsigned i = 0; i < length; ++i)
362 {
363 result[i] = js_array[i].as<double>();
364 }
365 return result;
366}

368// Helper function to convert a C++ vector to a JavaScript array
379val vector_to_array(const std::vector<double> &vec)
380{
381 val result = val::array();
382 for (size_t i = 0; i < vec.size(); ++i)
383 {
384 result.set(i, vec[i]);
385 }
386 return result;
387}
