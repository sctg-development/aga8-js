326// Helper function to convert a JavaScript array to a C++ vector
337std::vector<double> array_to_vector(const val &js_array)
338{
339 auto length = js_array["length"].as<unsigned>();
340 std::vector<double> result(length);
341 for (unsigned i = 0; i < length; ++i)
342 {
343 result[i] = js_array[i].as<double>();
344 }
345 return result;
346}

348// Helper function to convert a C++ vector to a JavaScript array
359val vector_to_array(const std::vector<double> &vec)
360{
361 val result = val::array();
362 for (size_t i = 0; i < vec.size(); ++i)
363 {
364 result.set(i, vec[i]);
365 }
366 return result;
