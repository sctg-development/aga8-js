27// The compositions in the gazMixtureInMolePercent array use the following order and must be sent as mole fractions:
28// 0 - PLACEHOLDER
29// 1 - Methane
30// 2 - Nitrogen
31// 3 - Carbon dioxide
32// 4 - Ethane
33// 5 - Propane
34// 6 - Isobutane
35// 7 - n-Butane
36// 8 - Isopentane
37// 9 - n-Pentane
38// 10 - n-Hexane
39// 11 - n-Heptane
40// 12 - n-Octane
41// 13 - n-Nonane
42// 14 - n-Decane
43// 15 - Hydrogen
44// 16 - Oxygen
45// 17 - Carbon monoxide
46// 18 - Water
47// 19 - Hydrogen sulfide
48// 20 - Helium
49// 21 - Argon
50//
51// For example, a mixture (in moles) of 94% methane, 5% CO2, and 1% helium would be (in mole fractions):
52// gazMixtureInMolePercent(1)=0.94, gazMixtureInMolePercent(3)=0.05, gazMixtureInMolePercent(20)=0.01
53enum class GasComponent
54{
55 PLACEHOLDER = 0,
56 METHANE = 1,
57 NITROGEN = 2,
58 CARBON_DIOXIDE = 3,
59 ETHANE = 4,
60 PROPANE = 5,
61 ISOBUTANE = 6,
62 N_BUTANE = 7,
63 ISOPENTANE = 8,
64 N_PENTANE = 9,
65 N_HEXANE = 10,
66 N_HEPTANE = 11,
67 N_OCTANE = 12,
68 N_NONANE = 13,
69 N_DECANE = 14,
70 HYDROGEN = 15,
71 OXYGEN = 16,
72 CARBON_MONOXIDE = 17,
73 WATER = 18,
74 HYDROGEN_SULFIDE = 19,
75 HELIUM = 20,
76 ARGON = 21
77};

156// Helper function to convert a JavaScript array to a C++ vector
157std::vector<double> array_to_vector(const val &js_array)
158{
159 auto length = js_array["length"].as<unsigned>();
160 std::vector<double> result(length);
161 for (unsigned i = 0; i < length; ++i)
162 {
163 result[i] = js_array[i].as<double>();
164 }
165 return result;
166}
167
168// Helper function to convert a C++ vector to a JavaScript array
169val vector_to_array(const std::vector<double> &vec)
170{
171 val result = val::array();
172 for (size_t i = 0; i < vec.size(); ++i)
173 {
174 result.set(i, vec[i]);
175 }
176 return result;
177}
